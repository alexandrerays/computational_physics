// subprogram que calcula o maior e o menor elemento de um vetor
void maior_menor(int N, int A[10], int *Maior, int *Menor){
  *Maior=A[0];
  *Menor=A[0];
  for(i=1;i<N;i++){
    if (A[i] > *Maior)  *Maior=A[i];
    if (A[i] < *Menor)  *Menor=A[i];
  }
  return;
}
